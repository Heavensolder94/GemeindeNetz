    function renderUserPosts() {
      const grid = document.getElementById("userPostsGrid");
      const user = state.currentUser;
      if (!user) return;

      const posts = getPosts()
        .filter(post => post.ownerId === user.id)
        .sort((a, b) => b.createdAt - a.createdAt);

      if (!posts.length) {
        grid.innerHTML = `
          <div class="empty-dashboard">
            <strong style="display:block;color:#30353d;font-size:1.05rem">Noch keine eigenen Beiträge.</strong>
            <span style="display:block;margin-top:8px">Erstelle dein erstes Angebot oder Gesuch direkt im Dashboard.</span>
          </div>
        `;
        return;
      }

      grid.innerHTML = posts.map(post => `
        <article class="post-card">
          <div class="post-image" style="background-image:url('${post.image}')"></div>
          <div class="post-body">
            <span class="post-type ${post.type === "search" ? "search" : ""}">
              ${post.type === "offer" ? "Wir bieten" : "Wir suchen"}
            </span>
            <h3>${escapeHtml(post.title)}</h3>
            <p>${escapeHtml(post.description)}</p>
            <div class="post-meta">
              <span>${escapeHtml(post.category)}</span>
              <span>${escapeHtml(post.location || user.location)}</span>
              <span>${escapeHtml(post.date || "Neu eingestellt")}</span>
            </div>
            <div class="post-actions">
              <button class="btn btn-ghost btn-small" onclick="showToast('Bearbeitungsfunktion als nächster Ausbauschritt.')">Bearbeiten</button>
              <button class="btn btn-danger btn-small" onclick="deletePost('${post.id}')">Löschen</button>
            </div>
          </div>
        </article>
      `).join("");
    }

    function deletePost(postId) {
      const user = state.currentUser;
      if (!user) return;
      const posts = getPosts();
      const post = posts.find(item => item.id === postId && item.ownerId === user.id);
      if (!post) return;

      const confirmed = confirm(`Beitrag „${post.title}“ wirklich löschen?`);
      if (!confirmed) return;

      savePosts(posts.filter(item => item.id !== postId));
      renderDashboard();
      showToast("Beitrag wurde gelöscht.");
    }

    window.deletePost = deletePost;

    function openPostModal() {
      if (!state.currentUser) {

      const backdrop = document.getElementById("postModalBackdrop");

      document.body.style.overflow = "";
    }

    function escapeHtml(value) {
      return String(value)
        .replaceAll("&", "&amp;")
        .replaceAll("<", "&lt;")
        .replaceAll(">", "&gt;")
        .replaceAll('"', "&quot;")
        .replaceAll("'", "&#039;");
    }

      toast.classList.add("show");
